QUICK START

Get Running in 5 Minutes

bash 
# Clone & Install
git clone https://github.com/your-org/nexus-agent
cd nexus-agent

# Backend Setup (Python 3.11+)
cd backend
pip install -r requirements.txt
cp .env.example .env
# Add AZURE_OPENAI_KEY, OPENAI_ENDPOINT, PINECONE_KEY

# Frontend Setup
cd ../frontend
npm install

# Start both services
npm run dev:full
python 
# Using the Python SDK
from nexus import AgentClient

agent = AgentClient(api_key="...")

# Run an autonomous task
result = agent.run(
  task="Find top 10 open source LLM projects on GitHub",
  mode="autonomous",
  max_steps=20,
  output_format="json"
)

print(result.data)       # Structured output
print(result.trace)      # Full execution trace
print(result.steps_used) # # of steps taken
